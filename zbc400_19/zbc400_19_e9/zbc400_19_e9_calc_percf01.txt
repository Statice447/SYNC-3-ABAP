FORM calc_perc  USING    p_gv_int1 TYPE i
                         p_gv_int2 TYPE i
                CHANGING p_gv_result TYPE tv_perc.

  DATA lv_pc TYPE p LENGTH 16 DECIMALS 3.

  lv_pc = p_gv_int1 * 100 / p_gv_int2.
  gv_result = lv_pc.

  WRITE: / 'lv_pc Result', lv_pc.

ENDFORM.
