MODULE exit INPUT.
  CASE ok_code.
    WHEN 'EXIT' OR 'CANC'.
      LEAVE PROGRAM.
  ENDCASE.

ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'DISPLAY'.
      PERFORM display_plant.
    WHEN 'CHANGE'.
      PERFORM change_plant.
    WHEN 'CREATE'.
      PERFORM create_plant.
    WHEN 'CLEAR'.
      CLEAR gs_plant_f.
    WHEN 'CHKDEL'.
      PERFORM check_plant.
  ENDCASE.

ENDMODULE.
