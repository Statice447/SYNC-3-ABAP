MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'DISPLAY'.
      PERFORM display_dvmt.
    WHEN 'CHANGE'.
      PERFORM change_dvmt.
    WHEN 'CREATE'.
      PERFORM create_dvmt.
    WHEN 'CLEAR'.
      CLEAR gs_dvmt_f.
    WHEN 'CELLCHECK'.
      PERFORM alv_cell.
  ENDCASE.

ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  EXIT  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE exit INPUT.
  CASE ok_code.
    WHEN 'EXIT' OR 'CANC'.
      LEAVE PROGRAM.
  ENDCASE.

ENDMODULE.
